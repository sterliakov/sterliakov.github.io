        \let\barone\pgfmathresult

        \node(Label) [
            text width=\barsize,
            yshift=-\currspacing,
            anchor=north west
        ] at (ActualBar.south -| 0,1) {##2};
        \node(FullWidth) [
            yshift=-4mm,
            draw=white, fill=gray, rectangle, minimum width=\barsize, minimum height=\barwidth,
            anchor=south west
        ] at (Label.south -| 0,1) {};
        \node(ActualBar) [
            yshift=-4mm,
            draw=white, fill=white, rectangle, minimum width=\barone, minimum height=\barwidth,
            anchor=south west
        ] at (Label.south -| 0,1) {};
        \ifthenelse{\equal{##1}{}}{
            \def\currspacing{2mm}
        }{
            \def\currspacing{-2mm}
            \node(ActualBar) [
                yshift=-1mm,
                anchor=north east,
            ] at (ActualBar.south -| \barone pt,1) {##1};
        }
    }
    \begin{tikzpicture}[node distance=5mm]
        \linespread{1}
        \node(ActualBar) [anchor=south west] at (0,0) {};
